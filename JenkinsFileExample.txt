pipeline {
    agent any
    stages {
        stage('Checkout') {
            steps {
                checkout([$class: 'GitSCM', branches: [[name: '*/main']], doGenerateSubmoduleConfigurations: false, extensions: [], submoduleCfg: [], userRemoteConfigs: [[credentialsId: 'jenkins', url: 'https://github.com/qaautomationlabs/playwright-jest.git']]])
            }
        }
        stage('Install Dependencies') {
            steps {
                sh 'npm install'
            }
        }
        stage('Run Tests') {
            steps {
                sh 'npm run test'
            }
        }
        stage('Publish Test Results') {
            steps {
                junit '**/test-results/*.xml'
            }
        }
        stage('Archive Artifacts') {
            steps {
                archiveArtifacts artifacts: '**/test-results/*.xml', fingerprint: true
            }
        }
        stage('Generate HTML Report') {
            steps {
                sh 'npx playwright show-report'
                archiveArtifacts artifacts: '**/report/*.html', fingerprint: true
            }
        }
    }
}
